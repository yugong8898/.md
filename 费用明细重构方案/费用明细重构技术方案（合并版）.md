### P0-1：countMoney 硬编码下标修复

**文件**：CostMessageDetails.vue | **改动量**：~5 行 | **修复 B1**

**现状**：L554~557，`countMoney` 通过 `this.newList[0]` 硬编码取第一行写入运费差价。
- L554-556：`confirmMoney` 有 `companyType !== 1` 守卫，仅普通运单执行
- L557：`confirmRemark` **无条件执行**，B1 企业时差价率写入货物保障服务行

当 `companyType=1`（B1 企业）时 `newList[0]` 是货物保障服务行，金额和差价率都写错位置。

**改法**：
```js
// 改前（L554~557）
if (this.detailObj.companyType !== 1) {
  this.$set(this.newList[0], 'confirmMoney', this.serviceRateMoney)
}
this.$set(this.newList[0], 'confirmRemark', rate)

// 改后
const freightDiffRow = this.newList.find(i => i.costName === '运费差价')

  this.$set(freightDiffRow, 'confirmMoney', this.serviceRateMoney)
```

### P0-2：debounce 写法修复

**文件**：CostMessageDetails.vue + edit.vue | **改动量**：~10 行 | **修复 B2**

**现状**：
- CostMessageDetails.vue L588：`debounce(this.dealZeroModeFn(), 500)` 传入了函数执行结果（undefined），debounce 完全无效

```js
// CostMessageDetails.vue：在 created 中预创建防抖函数
created() {

  }, 500)
}

// updBusCostInput 中
// [重构前] debounce(this.dealZeroModeFn(), 500)
this._debouncedDealZero()

// edit.vue L1103：改为直接调用（保持原有效行为：立即执行）
// [重构前] debounce(this.$refs.CostMessageDetails.dealZeroModeFn(), 500)

```

**为什么安全**：防抖只是延后执行，最终一定会执行一次；edit.vue 侧改为立即执行保持原有行为不变。

### P0-4：delTableRow 异步时序修复

**文件**：CostMessageDetails.vue | **改动量**：~5 行 | **修复 B4**

```js
// [重构前] delTableRow 末尾同步调用
delTableRow(index) {

  if ([101, 102].includes(row.orderNumber)) {

delTableRow(index) {
  // ...
  if ([101, 102].includes(row.orderNumber)) {

  } else {
```

### P0-5：commit() 重复编码校验修复

**文件**：CostMessageDetails.vue | **改动量**：~10 行 | **修复 B5**

**现状**：`commit()` 方法内用 `forEach` 遍历费用行，嵌套 `for...of map` 校验重复编码，发现重复时 `return false`。但 `return false` 对 `forEach` 无效，校验不通过时仍继续执行后续逻辑。最终 `n` 已正确赋值为 `'1'`/`'2'`，结尾 `return n` 符合契约——真正的瑕疵只是 `forEach` 内 `return false` 无效控制流（不影响正确性，但 `form.validate` 白跑一次）。

**⚠️ 关键约束**：`commit()` 的返回值契约是 `数组`（成功）或 `'1'`/`'2'`（失败）。`edit.vue` L1224 判断 `CostDetails === '1' || CostDetails === '2'`。**绝不能返回 `false`**，否则不等于 `'1'`/`'2'`，会走到 `changeMoney(false)` → `false.map()` 崩溃。

**改法**：
```js
// [重构前] forEach + return false（无效控制流，但 n 已赋值，行为正确）
this.params.model.forEach(res => {
  if (res.costType && res.costName) {
    // ... Map<costType, costCode[]> + Set 判重 ...
    for (const val of map) {
      if (new Set(val[1]).size !== val[1].length) {
        this.$message({ type: 'error', message: '相同费用类型和费用名称的费用项只允许录入一行' })
        n = '1'
        return false  // ← 只退出 forEach 回调，不退出 commit()
      }
    }
  } else {
    this.$message({ type: 'error', message: '费用类型或费用项目未填写完整' })
    n = '2'
  }
})

// 改后：用 labeled for...of + break outer 中断循环，保持 return n 契约
let n = ''
const map = new Map()
outer: for (const res of this.params.model) {
  if (res.costType && res.costName) {
    if (!map.has(res.costType)) map.set(res.costType, [res.costCode])
    else map.set(res.costType, [...map.get(res.costType), res.costCode])
    for (const val of map.values()) {
      if (new Set(val).size !== val.length) {
        this.$message({ type: 'error', message: '相同费用类型和费用名称的费用项只允许录入一行' })
        n = '1'
        break outer  // ← 跳出整个循环
      }
    }
  } else {
    this.$message({ type: 'error', message: '费用类型或费用项目未填写完整' })
    n = '2'
    break outer
  }
}
// ... 原有 validate / request 构建逻辑不变 ...
if (!n && this.params.model.length !== 0) return request
return n  // ← 保持 '1'/'2' 契约，绝不用 return false
```

```js
// 费用明细计算纯函数库，无副作用，可独立单测，不依赖 Vue 实例

import BigNumber from 'bignumber.js'
import { ZERO_ORDER_NUMBERS, PAYABLE_COST_TYPES } from '../config'

/**
 * 截取小数精度（保留 n 位有效小数，向下截断）
 */
export function getBit(value, n) {
  let str = value.toFixed(n + 1).toString()
  str = str.split('.')[0] + '.' + str.split('.')[1].substring(0, 7)
  return parseFloat(str)
}

/**
 * 按指定取整方式处理精度
 * @param {0|1|4} type 取整方式：0=向上，1=向下，4=四舍五入
 */
export function formatService(num, n, type) {
  const typeObj = { 0: Math.ceil, 1: Math.floor, 4: Math.round }
  return typeObj[type]((new BigNumber(num) * Math.pow(10, n)).toFixed(7)) / Math.pow(10, n)
}

/**
 * 运费差价核心公式：合计 = 小计 / (1 - 差价率%)
 * @param {number} amount 小计（司机运费确认金额）
 * @param {number} rate 差价率（百分比，如 10 表示 10%）
 * @param {0|1|4} roundingMode 取整方式
 */
export function count(amount, rate, roundingMode) {
  let num = amount / getBit((1 - (parseFloat(rate) / 100)), 7)
  num = new BigNumber(num).toFixed(7)
  return +formatService(num, 2, roundingMode || 0)
}

/**
 * 计算运费差价金额
 * 公式：运费差价 = count(小计, 差价率) - 小计 - 货运保障服务金额
 * @returns {string} 保留 2 位小数的字符串
 */
export function calcFreightDiff(subtotal, rate, premiumAmount, roundingMode) {
  const total = count(subtotal, rate, roundingMode) + premiumAmount
  return new BigNumber(total - subtotal - premiumAmount).toFixed(2)
}

/**
 * 按应付/应收方向累加费用列表中的某个金额字段
 * 统一消除组件内 4 处重复的 BigNumber reduce 模式
 *
 * @param {Array}   model             费用行数组（params.model 或 newList）
 * @param {string}  field             累加字段名（'updBusCost' | 'busCost' | 'confirmMoney'）
 * @param {boolean} [excludeZero]     是否排除抹零行（orderNumber 101/102），默认 false
 * @returns {BigNumber}               调用方按需 .toNumber() 或 .toFixed(2)
 */
export function calcNetAmount(model, field, { excludeZero = false } = {}) {
  return model.reduce((total, item) => {
    if (excludeZero && ZERO_ORDER_NUMBERS.includes(item.orderNumber)) return total
    const v = +(item[field] || 0)
    return PAYABLE_COST_TYPES.includes(`${item.costType}`) ? total.plus(v) : total.minus(v)
  }, new BigNumber(0))
}

/**
 * 计算第二表（汇总副表）合计金额
 * 与 calcNetAmount 的差异：油气优惠行做特殊减项，confirmMoney=-1 时按 0 处理
 */
export function calcSummaryTotal(newList) {
  return newList.reduce((total, item) => {
    if (item.costName === '油气优惠') {
      const v = Number(item.confirmMoney)
      return v === -1 ? total : total.minus(v) // -1 为未查到数据的占位值，不参与减项
    }
    return total.plus(+(item.confirmMoney || 0))
  }, new BigNumber(0))
}

// ── Vue mixin 导出（供 CostMessageDetails mixins 引用）────────────────────────
export default {
  methods: {
    getBit,
    formatService,
    count(Amount, Rate) {              // 薄代理：捕获组件的 roundingMode 上下文
      return count(Amount, Rate, this.roundingMode)
    },
    calcSubtotal(model) {              // 从 P1-3 直接平移，内部 reduce 替换为 calcNetAmount
      this.subtotal = calcNetAmount(model, 'updBusCost', { excludeZero: true }).toNumber()
    }
  }
}
```
