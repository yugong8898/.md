```javascript
// 运输里程格式化（假设后端返回米，转为公里）
formatTransportMileage(value) {
  return value != null ? new BigNumber(value).dividedBy(1000).toFixed(3) : '--'
}

// 车牌号批量输入规范化（英文逗号转空格）
normalizeBatchInput(value) {
  return typeof value === 'string' ? value.replace(/,/g, ' ').trim() : value
}
```
